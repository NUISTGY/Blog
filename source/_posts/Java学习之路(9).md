/**
 * 测试逻辑运算符和位运算符
 * @author 葛宇
 */
package 数据类型和运算符;

public class TestOperator_3 {
	public static void main(String[] args) {
		/*
		 * 与：&				同真则真，有假则假
		 * 或：|				同假则假，有真则真
		 * 非：！				真假颠倒
		 * 异或：^				同假异真
		 * 短路与：&&			有假则假
		 * 短路或：||				有真则真
		 * 
		 * 短路方式从左向右计算，若左边表达式计算结果足够判断则右边不进行表达式计算
		 */
		
		//////////////////////////////////////////////////
		
		/*
		 * 取反：~
		 * 按位与：&
		 * 按位或：|
		 * 按位异或：^
		 * 左移：<<				左移一位相当于乘2
		 * 右移：>>				右移一位相当于除2取商
		 */
		
		System.out.println(3<<2);          //相当于11左移两位变成1100，输出12
		System.out.println(3>>1);		   //相当于11右移一位变成1，输出1
	}
}
