```java
/**
 * Java的变量
 * @author 葛宇
 */
package 数据类型和运算符;

public class TestVariable {
	public static void main(String[] args) {
		int a;
		long b;
		double c;
		a=1;
		b=2;
		c=3;
	
		int i,j,k;
		i=1;
		j=2;
		k=3;
		i=j=k=4;
	    //i=5,j=6,k=7; 这种赋值Java只允许在变量定义时候进行
		
		int n=8,m=9;
	}
}

```
